let state={
  level:0,
  grid:[],    // current tile types

function computeTPX(){
  const maxW=Math.min(window.innerWidth-24,480);
  TPX=Math.floor(maxW/state.size);

// ── INIT LEVEL ──
function loadLevel(lvIdx){
  clearInterval(state.timerID);
  const lv=LEVELS[lvIdx];
  state.level=lvIdx;
  state.size=lv.size;
  state.moves=0;
  state.timer=0;
  state.solved=false;
  state.path=[];

  // deep copy grid
  state.grid=lv.grid.map(row=>[...row]);
  state.locked=new Set(lv.locked.map(({r,c})=>`${r},${c}`));

  computeTPX();
  shuffleNonLocked();
  renderGrid();
  updateHUD();
  startTimer();
  setMsg('Putar kepingan peta — sambungkan laluan kapal!');
  document.getElementById('level-badge').textContent=`PERINGKAT ${lvIdx+1}`;
  document.getElementById('sailing-ship').style.display='none';
  document.getElementById('overlay-win').style.display='none';
  document.getElementById('overlay-levels').style.display='none';
}

function shuffleNonLocked(){
  const lv=LEVELS[state.level];
  for(let r=0;r<state.size;r++){
    for(let c=0;c<state.size;c++){
      if(state.grid[r][c]==='.')continue;
      if(state.locked.has(`${r},${c}`))continue;
      // rotate randomly 0-3 times
      const times=Math.floor(Math.random()*4);
      for(let t=0;t<times;t++)state.grid[r][c]=ROTATE_MAP[state.grid[r][c]]||state.grid[r][c];
    }
  }
}

function shuffleLevel(){
  if(state.solved)return;
  shuffleNonLocked();
  state.moves=0;state.path=[];
  renderGrid();updateHUD();setMsg('Dikacau semula — cuba lagi!');
}

// ── RENDER ──
function renderGrid(){
  const grid=document.getElementById('grid');
  grid.style.gridTemplateColumns=`repeat(${state.size},${TPX}px)`;
  grid.innerHTML='';
  const lv=LEVELS[state.level];

  for(let r=0;r<state.size;r++){
    for(let c=0;c<state.size;c++){
      const tile=document.createElement('div');
      tile.className='tile';
      tile.style.width=TPX+'px';tile.style.height=TPX+'px';
      const key=`${r},${c}`;
      const type=state.grid[r][c];
      const isLocked=state.locked.has(key);
      const isStart=(lv.start.r===r&&lv.start.c===c);
      const isEnd=(lv.end.r===r&&lv.end.c===c);
      const onPath=state.path.some(p=>p.r===r&&p.c===c);

      if(isStart)tile.classList.add('start');
      if(isEnd)tile.classList.add('end');
      if(isLocked)tile.classList.add('locked');

      // draw pipe
      if(type&&type!=='.'){
        const cvs=document.createElement('canvas');
        cvs.width=TPX;cvs.height=TPX;
        drawPipe(cvs,type,onPath,isStart,isEnd);
        tile.appendChild(cvs);
      }

      // start/end icons
      if(isStart){
        const icon=document.createElement('div');
        icon.style.cssText=`position:absolute;inset:0;display:flex;align-items:center;justify-content:center;font-size:${TPX*.38}px;z-index:5;pointer-events:none;`;
        icon.textContent='⚓';
        tile.appendChild(icon);
      }
      if(isEnd){
        const icon=document.createElement('div');
        icon.style.cssText=`position:absolute;inset:0;display:flex;align-items:center;justify-content:center;font-size:${TPX*.38}px;z-index:5;pointer-events:none;`;
        icon.textContent='🏯';
        tile.appendChild(icon);
      }

      if(!isLocked&&type!=='.'){
        tile.addEventListener('click',()=>rotateTile(r,c));
      }
      grid.appendChild(tile);
    }
  }
}

// ── DRAW PIPE ON CANVAS ──
function drawPipe(cvs,type,onPath,isStart,isEnd){
  const ctx=cvs.getContext('2d');
  const w=cvs.width,h=cvs.height;
  const mid=w/2,thick=w*.18;
  ctx.clearRect(0,0,w,h);

  const color=onPath
    ?`rgba(40,180,255,0.9)`
    :`rgba(50,100,160,0.65)`;
  const glow=onPath?`rgba(80,200,255,0.4)`:null;

  ctx.lineCap='round';ctx.lineJoin='round';

  if(glow&&onPath){
    ctx.shadowBlur=10;ctx.shadowColor=glow;
  }

  ctx.strokeStyle=color;
  ctx.lineWidth=thick;

  const conns=PIPES[type]?.conns||[];
  // draw lines from center to each connection edge
  for(const dir of conns){
    ctx.beginPath();
    ctx.moveTo(mid,mid);
    if(dir===0)ctx.lineTo(mid,0);        // N
    else if(dir===1)ctx.lineTo(w,mid);   // E
    else if(dir===2)ctx.lineTo(mid,h);   // S
    else if(dir===3)ctx.lineTo(0,mid);   // W
    ctx.stroke();
  }

  // center dot
  ctx.shadowBlur=0;
  ctx.fillStyle=onPath?'rgba(100,220,255,0.9)':'rgba(60,120,200,0.7)';
  ctx.beginPath();ctx.arc(mid,mid,thick*.55,0,Math.PI*2);ctx.fill();

  // flow dots if on path
  if(onPath){
    ctx.fillStyle='rgba(200,240,255,0.7)';
    for(const dir of conns){
      const t=0.4;
      const dx=dir===1?w*t:dir===3?-w*t:0;
      const dy=dir===2?h*t:dir===0?-h*t:0;
      ctx.beginPath();ctx.arc(mid+dx,mid+dy,thick*.25,0,Math.PI*2);ctx.fill();
    }
  }
}

// ── ROTATE ──
function rotateTile(r,c){
  if(state.solved)return;
  if(state.locked.has(`${r},${c}`))return;
  if(state.grid[r][c]==='.')return;
  state.grid[r][c]=ROTATE_MAP[state.grid[r][c]]||state.grid[r][c];
  state.moves++;
  updateHUD();

  // check path
  state.path=solvePath();
  renderGrid();

  if(state.path.length>0&&checkSolved()){
    onSolved();
  } else {
    setMsg(state.moves===1?'Teruskan…':
      state.moves<5?'Hampir sampai?':
      state.moves<10?'Fikir dengan teliti…':
      'Sabar — setiap laluan ada penyelesaiannya');
  }
}

// ── PATH SOLVER (BFS along connected pipes) ──
function solvePath(){
  const lv=LEVELS[state.level];
  const start=lv.start;
  const end=lv.end;
  const visited=new Set();
  const queue=[{r:start.r,c:start.c,path:[{r:start.r,c:start.c}]}];

  while(queue.length){
    const {r,c,path}=queue.shift();
    const key=`${r},${c}`;
    if(visited.has(key))continue;
    visited.add(key);

    if(r===end.r&&c===end.c)return path;

    const type=state.grid[r][c];
    if(!type||type==='.')continue;
    const conns=PIPES[type]?.conns||[];

    for(const dir of conns){
      const [nr,nc,opp]=dir===0?[r-1,c,2]:dir===1?[r,c+1,3]:dir===2?[r+1,c,0]:[r,c-1,1];
      if(nr<0||nr>=state.size||nc<0||nc>=state.size)continue;
      const ntype=state.grid[nr][nc];
      if(!ntype||ntype==='.')continue;
      const nconns=PIPES[ntype]?.conns||[];
      if(nconns.includes(opp)&&!visited.has(`${nr},${nc}`)){
        queue.push({r:nr,c:nc,path:[...path,{r:nr,c:nc}]});
      }
    }
  }
  return [];
}

function checkSolved(){
  const lv=LEVELS[state.level];
  const end=lv.end;
  return state.path.length>0&&
    state.path[state.path.length-1].r===end.r&&
    state.path[state.path.length-1].c===end.c;
}

// ── WIN ──
function onSolved(){
  state.solved=true;
  clearInterval(state.timerID);
  animateShip();
  setTimeout(()=>{
    const lv=LEVELS[state.level];
    const mins=Math.floor(state.timer/60);
    const secs=state.timer%60;
    const rating=state.moves<=10?'S ⭐':state.moves<=18?'A':state.moves<=28?'B':'C';
    document.getElementById('win-desc').textContent=lv.story;
    document.getElementById('win-stats').textContent=
      `${state.moves} gerakan · ${mins}:${String(secs).padStart(2,'0')} masa · Penilaian ${rating}`;
    document.getElementById('overlay-win').style.display='flex';
  },1800);
  setMsg('🌊 Laluan terbuka! Armada Islam belayar maju!');
}

function animateShip(){
  const ship=document.getElementById('sailing-ship');
  const wrap=document.getElementById('grid-wrap');
  if(!state.path.length)return;
  ship.style.display='block';
  let idx=0;
  const step=()=>{
    if(idx>=state.path.length){ship.style.display='none';return;}
    const {r,c}=state.path[idx];
    ship.style.left=(c*TPX+TPX*.25)+'px';
    ship.style.top=(r*TPX+TPX*.18)+'px';
    ship.style.fontSize=(TPX*.45)+'px';
    idx++;
    setTimeout(step,200);
  };
  step();
}
